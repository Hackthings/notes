# automate setup

```
apt update
apt install ne
apt install -y zsh
chsh -s $(which zsh)
apt install curl
apt install git
sh -c "$(curl -fsSL https://raw.github.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"

```
